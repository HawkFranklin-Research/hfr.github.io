import pandas as pd
from lifelines import KaplanMeierFitter
from sklearn.ensemble import RandomForestClassifier

expr_data = pd.read_csv('tcga_aligned_expr.csv', index_col='PatientID')
surv_data = pd.read_csv('tcga_aligned_surv.csv', index_col='PatientID')

# Perform KMeans clustering
from sklearn.cluster import KMeans
kmeans = KMeans(n_clusters=8, random_state=42)
clusters = kmeans.fit_predict(expr_data)

# Survival analysis per cluster
kmf = KaplanMeierFitter()
for cluster in range(8):
    mask = clusters == cluster
    kmf.fit(surv_data['time'][mask], surv_data['event'][mask], label=f'Cluster {cluster}')
kmf.plot()
